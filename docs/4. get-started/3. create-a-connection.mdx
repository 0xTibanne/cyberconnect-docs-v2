### Connecting wallet
```jsx live
function ConnectButton() {
  // Check function if MetaMask is installed
  const isMetaMaskInstalled = () => {
    const { ethereum } = window;
    return Boolean(ethereum && ethereum.isMetaMask);
  }

  const handleOnClick = async () => {
    if(isMetaMaskInstalled()) {
      // Request to connect to MetaMask
      const accounts = await window.ethereum.request({
        method: "eth_requestAccounts"
      });
      alert(`Connected with: ${accounts[0]}`);
    } else {
      alert("Please install MetaMask.");
    }
  }
  
  return (
    <button
      className="connectButton"
      onClick={handleOnClick}
    >Connect wallet</button>
  );
}
```

### Create a follow button

Now you'll create a follow button in a couple of steps.

#### Step 1 - install SDK:

import Tabs from '@theme/Tabs';
import TabItem from '@theme/TabItem';

<Tabs>
  <TabItem value="npm" label="npm" default>
    npm install @cyberlab/cyberconnect
  </TabItem>
  <TabItem value="yarn" label="Yarn" default>
    yarn add @cyberlab/cyberconnect
  </TabItem>
</Tabs>

#### Step 2 - import it in app:
```jsx
import CyberConnect, {
  Env,
  Blockchain,
} from '@cyberlab/cyberconnect';

const cyberConnect = new CyberConnect({
  namespace: 'CyberConnect',
  env: Env.PRODUCTION,
  chain: Blockchain.ETH,
  provider: window.ethereum,
});
```

Now let's write the logic for the button..

```jsx live
function FollowButton() {
  const handleOnClick = async () => {
    // Prompt to enter the address
    const address = prompt("Enter the ens/address to follow:");

    // Create connection to address
    try {
      await cyberConnect.connect(address);
      alert(`Success: you're following ${address}!`);
    } catch (error) {
      alert(`Error: ${error.message}`);
    }
  }
  
  return (
    <button
      className="followButton"
      onClick={handleOnClick}
    >Follow address</button>
  );
}
```
