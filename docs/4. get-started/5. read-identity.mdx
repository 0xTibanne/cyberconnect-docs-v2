### Write the first query
Write the query

Make sure to check out the full list of fields available in the [CyberConnect API](/cyberconnect-api/overview/).

```jsx title="/src/queries/GetIdentity.js"
import { gql, useQuery } from "@apollo/client";
import { DEMO_ADDRESS } from "../helpers/constants";

export const GET_IDENTITY = gql`
    query($address: String!) {
        identity(address: $address) {
            domain
            social {
                twitter
            }
            followerCount
            followingCount
        }
    }
`;

export default function GetIdentity() {
    return useQuery(GET_IDENTITY, {
        variables: {
                address: DEMO_ADDRESS,
            },
        }).data;
}
```
